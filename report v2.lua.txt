local function createButton(text)
    local Frame = Instance.new("Frame", Scroll)
    Frame.Size = UDim2.new(1, -5, 0, 60)
    Frame.BackgroundColor3 = Color3.fromRGB(130,0,0)
    Frame.BorderSizePixel = 0
    Instance.new("UICorner", Frame).CornerRadius = UDim.new(0, 8)

    local Label = Instance.new("TextLabel", Frame)
    Label.Size = UDim2.new(0.65, 0, 1, 0)
    Label.Position = UDim2.new(0, 8, 0, 0)
    Label.BackgroundTransparency = 1
    Label.Text = text
    Label.TextColor3 = Color3.new(1,1,1)
    Label.Font = Enum.Font.Gotham
    Label.TextSize = 10
    Label.TextWrapped = true
    Label.TextXAlignment = Enum.TextXAlignment.Left
    Label.TextYAlignment = Enum.TextYAlignment.Top
    Label.TextTruncate = Enum.TextTruncate.AtEnd

    local Copy = Instance.new("TextButton", Frame)
    Copy.Size = UDim2.new(0, 60, 0, 24)
    Copy.Position = UDim2.new(1, -70, 0.5, -12)
    Copy.Text = "Copy"
    Copy.Font = Enum.Font.GothamBold
    Copy.TextSize = 11
    Copy.TextColor3 = Color3.new(1,1,1)
    Copy.BackgroundColor3 = Color3.fromRGB(200,0,0)
    Instance.new("UICorner", Copy).CornerRadius = UDim.new(0, 6)

    Label.MouseEnter:Connect(function()
        Tooltip.Text = text
        Tooltip.Position = UDim2.new(0, player:GetMouse().X + 10, 0, player:GetMouse().Y - 40)
        Tooltip.Visible = true
    end)

    Label.MouseLeave:Connect(function()
        Tooltip.Visible = false
    end)

    Copy.MouseButton1Click:Connect(function()
        ClickSound:Play()
        if setclipboard then
            setclipboard(text)
            Copy.Text = "✓"
            task.wait(0.8)
            Copy.Text = "Copy"
        end
    end)
end
